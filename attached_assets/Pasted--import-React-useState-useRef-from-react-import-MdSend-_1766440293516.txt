

import React, { useState, useRef } from 'react';
import { MdSend, MdAttachFile, MdClear } from 'react-icons/md';

interface MessageInputProps {
  onSendMessage: (text: string, files: File[]) => void;
  loading: boolean;
}

const MessageInput: React.FC<MessageInputProps> = ({ onSendMessage, loading }) => {
  const [inputText, setInputText] = useState('');
  const [attachedFiles, setAttachedFiles] = useState<File[]>([]);
  const fileInputRef = useRef<HTMLInputElement>(null);

  const handleSend = () => {
    if (inputText.trim() || attachedFiles.length > 0) {
      onSendMessage(inputText, attachedFiles);
      setInputText('');
      setAttachedFiles([]);
      if (fileInputRef.current) {
        fileInputRef.current.value = ''; // Clear file input field
      }
    }
  };

  const handleKeyDown = (event: React.KeyboardEvent<HTMLInputElement>) => {
    if (event.key === 'Enter' && !event.shiftKey && !loading) {
      event.preventDefault();
      handleSend();
    }
  };

  const handleFileChange = (event: React.ChangeEvent<HTMLInputElement>) => {
    if (event.target.files) {
      setAttachedFiles(Array.from(event.target.files));
    }
  };

  const removeFile = (index: number) => {
    setAttachedFiles(prev => prev.filter((_, i) => i !== index));
    if (fileInputRef.current && attachedFiles.length === 1) {
      fileInputRef.current.value = ''; // Clear file input if all removed
    }
  };

  return (
    <div className="p-4 bg-gray-100 border-t border-gray-300 flex flex-col md:flex-row items-center gap-2 sticky bottom-0 z-10">
      {attachedFiles.length > 0 && (
        <div className="flex flex-wrap gap-2 mb-2 md:mb-0 md:mr-2">
          {attachedFiles.map((file, index) => (
            <span key={index} className="flex items-center bg-blue-100 text-blue-800 text-xs px-2 py-1 rounded-full">
              {file.name}
              <button onClick={() => removeFile(index)} className="ml-1 text-blue-600 hover:text-blue-900">
                <MdClear size={14} />
              </button>
            </span>
          ))}
        </div>
      )}
      <div className="flex-grow flex w-full">
        <input
          type="text"
          className="flex-grow p-3 border border-gray-300 rounded-l-lg focus:outline-none focus:ring-2 focus:ring-blue-500 text-gray-800 bg-white"
          placeholder={loading ? "Esperando respuesta..." : "Escribe tu mensaje..."}
          value={inputText}
          onChange={(e) => setInputText(e.target.value)}
          onKeyDown={handleKeyDown}
          disabled={loading}
          aria-label="Escribir mensaje"
        />
        <label htmlFor="file-upload" className={`p-3 bg-gray-200 hover:bg-gray-300 text-gray-700 cursor-pointer transition-colors duration-200 ${loading ? 'opacity-50 cursor-not-allowed' : ''} flex items-center justify-center`}>
          <MdAttachFile size={24} />
          <span className="sr-only">Adjuntar archivo</span>
        </label>
        <input
          id="file-upload"
          type="file"
          ref={fileInputRef}
          multiple
          onChange={handleFileChange}
          className="hidden"
          disabled={loading}
          aria-label="Selector de archivos"
        />
        <button
          onClick={handleSend}
          className={`p-3 bg-blue-600 text-white rounded-r-lg hover:bg-blue-700 transition-colors duration-200 flex items-center justify-center ${loading ? 'opacity-50 cursor-not-allowed' : ''}`}
          disabled={loading}
          aria-label="Enviar mensaje"
        >
          {loading ? (
            <svg className="animate-spin h-5 w-5 text-white" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">
              <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4"></circle>
              <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>
            </svg>
          ) : (
            <MdSend size={24} />
          )}
        </button>
      </div>
    </div>
  );
};

export default MessageInput;
